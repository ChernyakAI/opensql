(
SELECT
TOP 1 ss.name+'.'+st.name [TableName]
FROM
sys.tables st (nolock)
INNER JOIN
sys.schemas ss (nolock)
ON
st.schema_id = ss.schema_id
GROUP BY
ss.name,st.name
ORDER BY
max(st.create_date) desc
)
